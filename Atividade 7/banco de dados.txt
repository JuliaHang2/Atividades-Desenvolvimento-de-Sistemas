create table contatos(
	id serial primary key,
	nome varchar(50),
	endereco varchar(50),
	telefone varchar(20),
	email varchar(50),
	celular varchar(20),
	imagem varchar(225),
	idfk int, FOREIGN KEY (idfk) references usuario (id)
	
);

create table usuario(

	id serial primary key,
	nomeUsuario varchar(50),
	senha varchar(50),
	email varchar(50),
	login varchar(50),
	imagemUsuario varchar(225),
	ativo boolean
);
